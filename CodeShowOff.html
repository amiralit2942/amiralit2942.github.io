public class MoveTransition : UiTransition
        {
            [SerializeField] private TransitionAction _transitionAction;
            [SerializeField] private TransitionActionMove _transitionActionMove;
            [SerializeField] private float _tweenTime = 0.4f;
            [SerializeField] private Image _darkBg;
    
            protected override void OnApply()
            {
                if (_transitionAction == TransitionAction.Close)
                    CloseSeq();
                else
                    OpenSeq();
            }
    
            private void CloseSeq()
            {
                var seq = DOTween.Sequence();
                switch (_transitionActionMove)
                {
                    case TransitionActionMove.Down:
                        seq.Append(_darkBg?.DOFade(0, _tweenTime / 2)).Append(rectTransform.DOMoveY(-650, _tweenTime))
                            .OnComplete(OnComplete);
                        break;
                    case TransitionActionMove.Left:
                        seq.Append(_darkBg?.DOFade(0, _tweenTime / 2)).Append(rectTransform.DOMoveX(-1920, _tweenTime))
                            .OnComplete(OnComplete);
                        break;
                    case TransitionActionMove.Right:
       
                        seq.Append(_darkBg?.DOFade(0, _tweenTime / 2)).Append(rectTransform.DOMoveX(1920, _tweenTime))
                            .OnComplete(OnComplete);
                        break;
                    case TransitionActionMove.Top:
                        seq.Append(_darkBg?.DOFade(0, _tweenTime / 2)).Append(rectTransform.DOMoveY(1686, _tweenTime))
                            .OnComplete(OnComplete);
                        break;
                }
            }
    
            private void OpenSeq()
            {
                switch (_transitionActionMove)
                {
                    case TransitionActionMove.Down:
                        rectTransform.anchoredPosition = new Vector2(0, -1500);
                        //seq.Append(rectTransform.DOMoveY(550, _tweenTime)).OnComplete(OnComplete);
                        break;
                    case TransitionActionMove.Left:
                        rectTransform.anchoredPosition = new Vector2(-1500, 0);
                        //seq.Append(rectTransform.DOMoveX(960, _tweenTime)).OnComplete(OnComplete);
                        break;
                    case TransitionActionMove.Right:
                        rectTransform.anchoredPosition = new Vector2(1500, 0);
                        // seq.Append(rectTransform.DOMoveX(960, _tweenTime)).OnComplete(OnComplete);
                        break;
                    case TransitionActionMove.Top:
                        rectTransform.anchoredPosition = new Vector2(0, 1500);
                        // seq.Append(rectTransform.DOMoveY(550, _tweenTime)).OnComplete(OnComplete);
                        break;
                }
    
                var seq = DOTween.Sequence();
                seq.Append(rectTransform.DOAnchorPos(new Vector2(0,0), _tweenTime)).OnComplete(OnComplete);
            }
        }
